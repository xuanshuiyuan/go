笔记：
string转成int：
int, _ := strconv.Atoi(string)

string转成int64：
int64, _:= strconv.ParseInt(string, 10, 64)

int转成string：
string := strconv.Itoa(int)

int64转成string：
string := strconv.FormatInt(int64,10)

float64转成string：
string := strconv.FormatFloat(float64, 'f', -1, 64)
// 'b' (-ddddp±ddd，二进制指数)
// 'e' (-d.dddde±dd，十进制指数)
// 'E' (-d.ddddE±dd，十进制指数)
// 'f' (-ddd.dddd，没有指数)
// 'g' ('e':大指数，'f':其它情况)
// 'G' ('E':大指数，'f':其它情况)
